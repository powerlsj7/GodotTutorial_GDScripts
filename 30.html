randf함수는 0~1의 부동소수점 값을 반환함. 0.8일때와 아닐 때 출력을 나눠서 확인해봄.

extends Node

func _ready():
	#0~1의 값을 반환하는함수 randf
	for i in 10:
		var roll=randf()
		if roll <=0.8:
			print("You found a COMMON item.")
		else:
			print("You found a Rare item!")
