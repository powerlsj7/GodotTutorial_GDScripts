배열 요소에 접근하는 것과 수정(제거, 추가)하는 방법

extends Node

func _ready():
	#대괄호안에 요소를 넣을 수 있음.
	#요소는 자료형이 달라도 가능함
	#만약 요소를 특정 자료형으로 하고 싶다면
	var item:Array[String]=["Potion","Feather", "Stolen harp"]
	#요소에 접근하려면 [숫자0~크기] 로 접근가능함
	print(item[0])
	print(item[1])
	print(item[2])
	#수정도 가능함 
	item[1]="이게 1번?"
	item[2]="변함!"
	print(item[0])
	print(item[1])
	print(item[2])
	# 번호로 제거도 가능함
	item.remove_at(1)
	#추가도 가능함
	item.append("추가")
	print(item[0])
	print(item[1])
	print(item[2])
