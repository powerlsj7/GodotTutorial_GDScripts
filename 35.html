에디터의 노드 트리에 보이는 노드를 스크립트로 불러오는 방법.

드래그로 경로만 불러오거나 Ctrl을 같이 눌러서 드래그하면 @Onready가 붙으면서 자동으로 변수가 생긴다.

@Onready는 Gau func _ready함수가 실행되기 전에 초기화된다.

extends Node
#왼쪽 노드트리에서 특정 노드를 참조하는 법을 배움.
#마우스 드래그로 스크립트창에 불러올 수 있음.
#$Player/Weapon
#Ctrl을 누르면 앞에 @onready가 붙음.
#@onready var weapon_2 = $Player/Weapon2
#위 문법은 아래 문법과 동일하다.
@onready var Weapon=get_node( "Player/Weapon")
#Onready는 _ready함수가 작동하기 직전에 초기화한다.
func _ready():
	# 변수의 경로를 확인할 수 있다.
	print(Weapon.get_path())

@export로 에디터 인스펙터에서 참조가 가능하다

extends Node
#왼쪽 노드트리에서 특정 노드를 참조하는 법을 배움.
#마우스 드래그로 스크립트창에 불러올 수 있음.
#$Player/Weapon
#Ctrl을 누르면 앞에 @onready가 붙음.
#@onready var weapon_2 = $Player/Weapon2
#위 문법은 아래 문법과 동일하다.
#@export를 사용하여 에디터 우측 인스펙터에서 참조하는 것도 가능하다.
@export var Weapon :Node
#Onready는 _ready함수가 작동하기 직전에 초기화한다.
func _ready():
	print(Weapon.get_path())

func _ready():
	if Weapon is Node2D:
