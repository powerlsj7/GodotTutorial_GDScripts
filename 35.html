match문을 배워봄. switch랑 비슷하지만 좀더 강력함.

자료형값이 맞는지 정확히 비교함

extends Node
#이번에는 게임에서 상태나 태그를 지정하는 enum을 배워봄.
enum Alignment{ALLY=0, NEUTRAL=1,ENEMY=-1}
#사실상 enum은 초기화한 뒤 번호를 지정하지 않으면 0부터 쭉 순서대로 부여됨.
@export var my_alignment : Alignment
func _ready():
	#match문을 배워봄.
	#switch문이랑 비슷하지만 좀더 강력함.
	#자료형을 정확하게 인식함. 정수 1과 1.0은 다른걸로 판단.
	match my_alignment:
		Alignment.ALLY:
			print("Hello, friend!")
		Alignment.NEUTRAL:
			print("I come in piece!")
		Alignment.ENEMY:
			print("TASTE MY WRATH!")
		_:#이건 기본값으로 위의 상태값이 아니면 이 부분이 실행됨.
			print("Who art thou?")
	#enum이 아니어도 사용가능
	var test=1.0
	match test:
		1.0:
			print("float")
		1:
			print(1)
		_:
			print("기본")
