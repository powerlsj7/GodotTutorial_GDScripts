만든 배열을 활용하여 반복실행 문법을 배워봄.

다음과 같은 문법으로 배열의 요소 i에 접근할 수 있음.

각 반복실행문에서 조건(if) 등을 걸수도 있음.

extends Node

func _ready():
	#만든 배열로 반복문에 활용
	var item:Array[String]=["Potion","Feather", "Stolen harp"]
	for i in item:
		print(i)
	for i in item:
		if i.length()>6:
			print(i)

for문에 숫자를 넣고 총 몇번 반복되는지 확인할 수 있음.

while문도 배워봄. while문에 조건을 걸고 랜덤 값으로 물을 채움.유리잔이 반이 넘어가면 종료되도록 작성.

extends Node

func _ready():
	#이번에는 n(0)~8-1 까지 출력을 확인.
	for n in 8:
		print(n)
	#while문으로 반복실행가능
	var glass:=0.0
	while glass<0.5:
		glass+=randf_range(0.01,0.2)
		print(glass)
	print("The glass is now half full")

glass+=randf_range(0.01,0.2) 이부분이 지워진다면 조건에 들어올수 없고 무한루프에 빠질 수 있으니 주의가 필요함.

continue와 break은 직접 사용해보고 판단.

extends Node

func _ready():
	#이번에는 n(0)~8-1 까지 출력을 확인.
	for n in 8:
		print(n)
	#while문으로 반복실행가능
	var glass:=0.0
	while glass<0.5:
		glass+=randf_range(0.01,0.2)
		#continue는 조건이 다음과 같이 있다면
		# 다음 실행문 print("continue안걸림") 으로 진행되지 않고 처음구간으로 돌아감.
		if glass <0.2:
			continue
		print("continue안걸림")
	#break으로 조건을 걸어 종료하거나 바로 종료하고 while문을 빠져나올 수 있음.
		if glass >0.2:
			print("break으로 종료")
			break
	print("The glass is now half full")
